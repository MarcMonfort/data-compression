156  public int[][][] blockSplitting(int[] vector) {
157  int[][][] mcuBlocks = new int[bloquesHorizontales*bloquesVerticales][BLOC_SIZE][BLOC_SIZE];
158  // Recorrido = O(n*m)
159  /* C√°lculo √≠ndice vector = ((h*BLOC_SIZE + hb) * width) + (w*BLOC_SIZE + wb) */
160  int indiceVector = 0;
161  for (int h = 0; h < bloquesVerticales; h++) {
162  int maxI = (!hRegular && h == bloquesVerticales-1) ? height % BLOC_SIZE : BLOC_SIZE;
163 
164  for (int hb = 0; hb < maxI; hb++) {
165  // Recorrido sobre el vector que tendr√° tantos width como height. w0,w1..wh (h=height)
166  // Se recorre por todos los bloques y luego cada bloque por separado
167 
168  for (int w = 0; w < bloquesHorizontales; w++) {
169  int maxJ = (!wRegular && w == bloquesHorizontales-1) ? width % BLOC_SIZE : BLOC_SIZE;
170  for (int wb = 0; wb < maxJ; wb++) {
171  mcuBlocks[h*bloquesHorizontales + w][hb][wb] = vector[indiceVector++];
172  }
173  if (!wRegular && w == bloquesHorizontales - 1) {
174  for (int wb = maxJ; wb < BLOC_SIZE; wb++) {
175  // Rellena con los bordes
176  mcuBlocks[h*bloquesHorizontales + w][hb][wb] = vector[indiceVector - 1];
177  }
178  }
179  }
180  }
181  if (!hRegular && h == bloquesVerticales - 1) {
182  for (int hb = maxI; hb < BLOC_SIZE; hb++) {
183  // Rellena con los bordes en la parte de abajo
184  for (int w = 0; w < bloquesHorizontales; w++) {
185  for (int wb = 0; wb < BLOC_SIZE; wb++) {
186  mcuBlocks[h*bloquesHorizontales + w][hb][wb] = mcuBlocks[h*bloquesHorizontales + w][maxI - 1][wb];
